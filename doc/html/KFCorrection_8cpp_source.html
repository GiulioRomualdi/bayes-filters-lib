31 std::pair<bool, VectorXd> KFCorrection::getLikelihood()
32 {
33  if ((innovations_.rows() == 0) || (innovations_.cols() == 0))
34  return std::make_pair(false, VectorXd());
35 
36  VectorXd likelihood(innovations_.cols());
37  for (std::size_t i = 0; i < likelihood.size(); i++)
38  {
39  likelihood(i) = utils::multivariate_gaussian_density(innovations_.col(i), VectorXd::Zero(innovations_.rows()), meas_covariances_.covariance(i)).coeff(0);
40  }
41 
42  return std::make_pair(true, likelihood);
43 }

46 void KFCorrection::correctStep(const GaussianMixture& pred_state, GaussianMixture& corr_state)
47 {
48  /* Get the current measurement if available. */
49  bool valid_measurement;
50  Data measurement;
51  std::tie(valid_measurement, measurement) = measurement_model_->measure();
52 
53  if (!valid_measurement)
54  {
55  corr_state = pred_state;
56  return;
57  }
58 
59  /* Evaluate the predicted measurement if possible. */
60  bool valid_predicted_measurement;
61  Data predicted_measurement;
62  std::tie(valid_predicted_measurement, predicted_measurement) = measurement_model_->predictedMeasure(pred_state.mean());
63 
64  if (!valid_predicted_measurement)
65  {
66  corr_state = pred_state;
67  return;
68  }
69 
70  /* Evaluate the innovation if possible. */
71  bool valid_innovation;
72  Data innovation;
73  std::tie(valid_innovation, innovation) = measurement_model_->innovation(predicted_measurement, measurement);
74 
75  if (!valid_innovation)
76  {
77  corr_state = pred_state;
78  return;
79  }
80 
81  bool valid_covariance_matrix;
82  MatrixXd R;
83  std::tie(valid_covariance_matrix, R) = measurement_model_->getNoiseCovarianceMatrix();
84 
85  if (!valid_covariance_matrix)
86  {
87  corr_state = pred_state;
88  return;
89  }
90 
91  MatrixXd H = measurement_model_->getMeasurementMatrix();
92 
93  /* Cast innovations once for all. */
94  innovations_ = any::any_cast<MatrixXd&&>(std::move(innovation));
95 
96  /* Initialize measurement covariances.
97  GaussianMixture will effectively resize only if it needs to. */
98  meas_covariances_.resize(pred_state.components, H.rows());
99 
100  /* Process all the components in the mixture. */
101  for (size_t i = 0; i < pred_state.components; i++)
102  {
103  /* Evaluate the measurement covariance matrix
104  Py = H * Px * H' + R */
105  meas_covariances_.covariance(i) = H * pred_state.covariance(i) * H.transpose() + R;
106 
107  /* Evaluate the Kalman Gain
108  K = Px * H' * (Py)^{-1} */
109  MatrixXd K = pred_state.covariance(i) * H.transpose() * meas_covariances_.covariance(i).inverse();
110 
111  /* Evaluate the filtered mean
112  x_{k}+ = x{k}- + K * (y - y_predicted) */
113  corr_state.mean(i) = pred_state.mean(i) + K * innovations_.col(i);
114 
115  /* Evaluate the filtered covariance
116  P_{k}+ = P_{k}- - K * Py * K' */
117  corr_state.covariance(i).noalias() = pred_state.covariance(i) - K * meas_covariances_.covariance(i) * K.transpose();
118  }
119 }
