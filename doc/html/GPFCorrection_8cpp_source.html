98 void GPFCorrection::correctStep(const bfl::ParticleSet& pred_particles, bfl::ParticleSet& corr_particles)
99 {
100  /* Propagate Gaussian belief associated to each particle. */
101  gaussian_correction_->correct(pred_particles, corr_particles);
102 
103  /* Sample from the proposal distribution. */
104  for (std::size_t i = 0; i < pred_particles.components; i++)
105  {
106  corr_particles.state(i) = sampleFromProposal(corr_particles.mean(i), corr_particles.covariance(i));
107  }
108 
109  /* Evaluate the likelihood. */
110  std::tie(valid_likelihood_, likelihood_) = getLikelihoodModel().likelihood(getMeasurementModel(), corr_particles.state());
111 
112  if (!valid_likelihood_)
113  {
114  corr_particles = pred_particles;
115 
116  return;
117  }
118 
119  /* Evaluate the transition probability. */
120  VectorXd transition_probability = state_model_->getTransitionProbability(pred_particles.state(), corr_particles.state());
121 
122  /* Update weights in the log space.
123  * w_{k} = w_{k-1} + log(likelihood) + log(transition_probability) - log(proposal_distribution)
124  */
125  double eps = std::numeric_limits<double>::min();
126  for (std::size_t i = 0; i < pred_particles.components; i++)
127  {
128  corr_particles.weight(i) = pred_particles.weight(i) + std::log(likelihood_(i) + eps) + std::log(transition_probability(i) + eps) - std::log(evaluateProposal(corr_particles.state(i), corr_particles.mean(i), corr_particles.covariance(i)) + eps);
129  }
130 }

133 Eigen::VectorXd GPFCorrection::sampleFromProposal(const Eigen::VectorXd& mean, const Eigen::MatrixXd& covariance)
134 {
135  /* Evaluate the square root of the state covariance matrix using the LDL' decomposition
136  (it can be used even if the covariance matrix is positive semidefinite). */
137  LDLT<MatrixXd> chol_ldlt(covariance);
138  MatrixXd sqrt_P = (chol_ldlt.transpositionsP() * MatrixXd::Identity(mean.size(), mean.size())).transpose() *
139  chol_ldlt.matrixL() *
140  chol_ldlt.vectorD().real().cwiseSqrt().asDiagonal();
141 
142  /* Sample i.i.d standard normal univariates. */
143  VectorXd rand_vectors(mean.size());
144  for (int i = 0; i < rand_vectors.size(); i++)
145  rand_vectors(i) = gaussian_random_sample_();
146 
147  /* Return a sample from a normal multivariate having mean `mean` and covariance 'covariance'. */
148  return mean + sqrt_P * rand_vectors;
149 }
