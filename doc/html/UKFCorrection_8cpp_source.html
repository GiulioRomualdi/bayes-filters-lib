71 std::pair<bool, VectorXd> UKFCorrection::getLikelihood()
72 {
73  if ((innovations_.rows() == 0) || (innovations_.cols() == 0))
74  return std::make_pair(false, VectorXd());
75 
76  VectorXd likelihood(innovations_.cols());
77  for (std::size_t i = 0; i < innovations_.cols(); i++)
78  {
79  likelihood(i) = utils::multivariate_gaussian_density(innovations_.col(i), VectorXd::Zero(innovations_.rows()), predicted_meas_.covariance(i)).coeff(0);
80  }
81 
82  return std::make_pair(true, likelihood);
83 }

86 void UKFCorrection::correctStep(const GaussianMixture& pred_state, GaussianMixture& corr_state)
87 {
88  /* Pick the correct measurement model. */
89  MeasurementModel& model = getMeasurementModel();
90 
91  /* Get the current measurement if available. */
92  bool valid_measurement;
93  Data measurement;
94  std::tie(valid_measurement, measurement) = model.measure();
95 
96  if (!valid_measurement)
97  {
98  corr_state = pred_state;
99  return;
100  }
101 
102  /* Extract measurement size. */
103  std::size_t meas_size = model.getMeasurementDescription().total_size();
104 
105  /* Evaluate the joint state-measurement statistics, if possible. */
106  bool valid = false;
107  MatrixXd Pxy;
108  if (type_ == UKFCorrectionType::Generic)
109  {
110  /* Augment the previous state using measurement noise statistics. */
111  GaussianMixture pred_state_augmented = pred_state;
112 
113  MatrixXd noise_covariance_matrix;
114  std::tie(std::ignore, noise_covariance_matrix) = model.getNoiseCovarianceMatrix();
115  pred_state_augmented.augmentWithNoise(noise_covariance_matrix);
116 
117  if (update_weights_online_)
118  ut_weight_ = UTWeight(measurement_model_->getInputDescription(), ut_alpha_, ut_beta_, ut_kappa_);
119 
120  std::tie(valid, predicted_meas_, Pxy) = sigma_point::unscented_transform(pred_state_augmented, ut_weight_, *measurement_model_);
121  }
122  else if (type_ == UKFCorrectionType::Additive)
123  {
124  std::tie(valid, predicted_meas_, Pxy) = sigma_point::unscented_transform(pred_state, ut_weight_, *additive_measurement_model_);
125  }
126 
127  if (!valid)
128  {
129  corr_state = pred_state;
130  return;
131  }
132 
133  /* Evaluate the innovation if possible. */
134  bool valid_innovation;
135  Data innovation;
136  /* This temporary is required since some MeasurementModel::innovation methods may try to cast from
137  const Ref<const MatrixXd> to MatrixXd resulting in a bfl::any::bad_any_cast.
138 
139  Hopefully, using std::move, it is possible to steal the memory from predicted_meas_.mean(). */
140  MatrixXd y_p = std::move(predicted_meas_.mean());
141  std::tie(valid_innovation, innovation) = model.innovation(y_p, measurement);
142 
143  if (!valid_innovation)
144  {
145  corr_state = pred_state;
146  return;
147  }
148 
149  /* Cast innovations once for all. */
150  innovations_ = any::any_cast<MatrixXd&&>(std::move(innovation));
151 
152  /* Process all the components in the mixture. */
153  for (size_t i=0; i < pred_state.components; i++)
154  {
155  /* Evaluate the Kalman Gain
156  K = Pxy * (Py)^{-1} */
157  MatrixXd K = Pxy.middleCols(meas_size * i, meas_size) * predicted_meas_.covariance(i).inverse();
158 
159  /* Evaluate the filtered mean.
160  x_{k}+ = x{k}- + K * innovation */
161  corr_state.mean(i) = pred_state.mean(i) + K * innovations_.col(i);
162 
163  /* Evaluate the filtered covariance
164  P_{k}+ = P_{k}- - K * Py * K' */
165  corr_state.covariance(i) = pred_state.covariance(i) - K * predicted_meas_.covariance(i) * K.transpose();
166  }
167 }
