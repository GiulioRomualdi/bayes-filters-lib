103 {
104  do

108 
109  std::unique_lock<std::mutex> lk(mtx_run_);
110  cv_run_.wait(lk, [this]{ return (this->run_ || this->teardown_); });

112  {
113  lk.unlock();
114  }
115  catch (const std::system_error& e)
116  {

120  teardown_ = true;
121  }
122 
123  initialization_step();
124

126  {

131  }
